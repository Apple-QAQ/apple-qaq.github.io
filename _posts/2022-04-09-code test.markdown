import turtle as t
import random
import time


# 新建turtle实例

t0 = t.Turtle()
t0.color("black")
t0.shape("turtle")
t0.pu()
t0.goto(-160,100)

t1 = t.Turtle()
t1.color("red")
t1.shape("turtle")
t1.pu()
t1.goto(-160,50)

t2 = t.Turtle()
t2.color("blue")
t2.shape("turtle")
t2.pu()
t2.goto(-160,0)

t3 = t.Turtle()
t3.color("green")
t3.shape("turtle")
t3.pu()
t3.goto(-160,-50)

t.pu()
t.goto(-130,110)
t.hideturtle()
t.speed(8)
for j in range(14):
    t.write(j)
    t.pu()
    t.right(90)
    t.forward(10)
    t.pd()
    t.forward(160)
    t.backward(170)
    t.pu()
    t.left(90)
    t.fd(20)

# sum:
d_sum = 0
d1_sum = 0
d2_sum = 0
d3_sum = 0

# time:

    if d_sum <= 280:

        t1.forward(d1)
        d1_sum += d1
